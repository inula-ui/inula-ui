The simplest usage.

# zh-CN

最简单的用法。

```tsx
import { Button, Dropdown, Icon, Separator } from '@inula-ui/components';
import KeyboardArrowDownOutlined from '@material-design-icons/svg/outlined/keyboard_arrow_down.svg?react';
import LogoutOutlined from '@material-design-icons/svg/outlined/logout.svg?react';

export default function Demo() {
  return (
    <div className="row g-2">
      <div className="col-auto">
        <Dropdown
          list={[
            { id: 'Item1', title: 'Item 1', type: 'item' },
            { id: 'Item2', title: 'Item 2', type: 'item' },
            {
              id: 'Item3',
              title: 'Item 3',
              type: 'item',
              icon: (
                <Icon>
                  <LogoutOutlined />
                </Icon>
              ),
              disabled: true,
            },
          ]}
        >
          <Button
            pattern="secondary"
            icon={
              <Icon>
                <KeyboardArrowDownOutlined />
              </Icon>
            }
            iconRight
          >
            Hover me
          </Button>
        </Dropdown>
      </div>
      <div className="col-auto">
        <Dropdown
          list={[
            { id: 'Item1', title: 'Item 1', type: 'item' },
            { id: 'Item2', title: 'Item 2', type: 'item' },
            {
              id: 'Group3',
              title: 'Group 3',
              type: 'group',
              children: [
                { id: 'Item31', title: 'Item 31', type: 'item' },
                { id: 'Item32', title: 'Item 32', type: 'item', disabled: true },
              ],
            },
          ]}
          trigger="click"
          popupRender={(el) => (
            <>
              {el}
              <Separator style={{ margin: 0 }} />
              <div style={{ padding: '8px 12px' }}>
                <Button size="small">Click me</Button>
              </div>
            </>
          )}
        >
          <Button
            pattern="secondary"
            icon={
              <Icon>
                <KeyboardArrowDownOutlined />
              </Icon>
            }
            iconRight
          >
            Click me
          </Button>
        </Dropdown>
      </div>
    </div>
  );
}
```

# zh-CN

这里展示了如何自定义操作。

```tsx
import { Button, Checkbox, Icon, Radio, Separator, Table } from '@inula-ui/components';
import { useImmer } from '@inula-ui/hooks';
import FilterAltFilled from '@material-design-icons/svg/filled/filter_alt.svg?react';
import SearchOutlined from '@material-design-icons/svg/outlined/search.svg?react';
import { Children } from 'openinula';

const origin = [1, 3, 2].map((n) => ({
  name: `Name ${6 - n}`,
  age: `2${n}`,
  address: `Room 100${n}, Building 2, Fandou Garden, Fandou Street`,
}));
export default function Demo() {
  const [sort, setSort] = useImmer({ age: 'descend' });
  const [search, setSearch] = useImmer({
    age: '',
  });
  const [filter, setFilter] = useImmer({
    _name: null,
    _age: [],
    _address: '',
    name: null,
    age: [],
    address: '',
  });

  const sortFn = (() => {
    if (sort.name) {
      return (a, b) => a.name.localeCompare(b.name) * (sort.name === 'ascend' ? 1 : -1);
    }
    if (sort.age) {
      return (a, b) => a.age.localeCompare(b.age) * (sort.age === 'ascend' ? 1 : -1);
    }
  })();
  const list = origin.filter((data) => {
    if (filter.name !== null && data.name !== filter.name) {
      return false;
    }
    if (filter.age.length > 0 && !filter.age.includes(data.age)) {
      return false;
    }
    if (filter.address && !data.address.includes(filter.address)) {
      return false;
    }
    return true;
  });
  if (sortFn) {
    list.sort(sortFn);
  }

  return (
    <Table style={{ minWidth: 600 }}>
      <table>
        <thead>
          <tr>
            <Table.Th
              sort={{
                options: ['descend', null],
                active: sort.name ?? null,
                onChange: (order) => {
                  setSort({ name: order });
                },
              }}
              action={
                <Table.Filter
                  content={
                    <Radio.Group
                      list={Array.from({ length: 6 }).map((_, i) => ({ label: `Name ${i + 1}`, value: `Name ${i + 1}` }))}
                      model={filter._name}
                      onModelChange={(value) => {
                        setFilter((draft) => {
                          draft._name = value;
                        });
                      }}
                    >
                      {(nodes) => (
                        <div className="d-flex flex-column" style={{ maxHeight: 168, overflow: 'auto' }}>
                          {Children.map(nodes, (node) => (
                            <div className="d-flex align-items-center" style={{ minHeight: 28 }}>
                              {node}
                            </div>
                          ))}
                        </div>
                      )}
                    </Radio.Group>
                  }
                  onReset={() => {
                    setFilter((draft) => {
                      draft._name = null;
                    });
                  }}
                  onVisibleChange={(visible) => {
                    if (!visible) {
                      setFilter((draft) => {
                        draft.name = draft._name;
                      });
                    }
                  }}
                >
                  <Table.ThAction active={!!filter.name}>
                    <Icon>
                      <FilterAltFilled />
                    </Icon>
                  </Table.ThAction>
                </Table.Filter>
              }
            >
              Name
            </Table.Th>
            <Table.Th
              sort={{
                active: sort.age ?? null,
                onChange: (order) => {
                  setSort({ age: order });
                },
              }}
              action={
                <Table.Filter
                  content={
                    <Checkbox.Group
                      list={Array.from({ length: 100 })
                        .map((_, i) => ({ label: `${i}`, value: `${i}` }))
                        .filter(({ label }) => label.includes(search.age))}
                      model={filter._age}
                      onModelChange={(value) => {
                        setFilter((draft) => {
                          draft._age = value;
                        });
                      }}
                    >
                      {(nodes) => (
                        <div className="d-flex flex-column mt-2" style={{ maxHeight: 168, overflow: 'auto' }}>
                          {Children.map(nodes, (node) => (
                            <div className="d-flex align-items-center" style={{ minHeight: 28 }}>
                              {node}
                            </div>
                          ))}
                        </div>
                      )}
                    </Checkbox.Group>
                  }
                  searchable
                  searchValue={search.age}
                  onSearch={(value) => {
                    setSearch((draft) => {
                      draft.age = value;
                    });
                  }}
                  onReset={() => {
                    setFilter((draft) => {
                      draft._age = [];
                    });
                  }}
                  onVisibleChange={(visible) => {
                    if (!visible) {
                      setFilter((draft) => {
                        draft.age = draft._age;
                      });
                    }
                  }}
                >
                  <Table.ThAction active={filter.age.length > 0}>
                    <Icon>
                      <FilterAltFilled />
                    </Icon>
                  </Table.ThAction>
                </Table.Filter>
              }
            >
              Age
            </Table.Th>
            <Table.Th
              action={
                <Table.Filter
                  searchable
                  searchValue={filter._address}
                  onSearch={(value) => {
                    setFilter((draft) => {
                      draft._address = value;
                    });
                  }}
                  onReset={() => {
                    setFilter((draft) => {
                      draft._address = '';
                    });
                  }}
                  onVisibleChange={(visible) => {
                    if (!visible) {
                      setFilter((draft) => {
                        draft.address = draft._address;
                      });
                    }
                  }}
                >
                  <Table.ThAction active={filter.address.length > 0}>
                    <Icon>
                      <SearchOutlined />
                    </Icon>
                  </Table.ThAction>
                </Table.Filter>
              }
            >
              Address
            </Table.Th>
            <Table.Th>Action</Table.Th>
          </tr>
        </thead>
        <tbody>
          {list.length === 0 ? (
            <Table.Empty />
          ) : (
            list.map((data) => (
              <tr key={data.name}>
                <Table.Td>{data.name}</Table.Td>
                <Table.Td>{data.age}</Table.Td>
                <Table.Td>{data.address}</Table.Td>
                <Table.Td>
                  <div className="d-flex">
                    <Button pattern="link">Edit</Button>
                    <Separator vertical />
                    <Button pattern="link">Delete</Button>
                  </div>
                </Table.Td>
              </tr>
            ))
          )}
        </tbody>
      </table>
    </Table>
  );
}
```
